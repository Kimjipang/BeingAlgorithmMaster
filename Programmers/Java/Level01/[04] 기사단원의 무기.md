// 풀이 1

class Solution {
    public int solution(int number, int limit, int power) {
        int answer = 0;
        int[] knights = new int[number];
        
        for(int i = 1; i <= knights.length; i++) {
            int count = 0;
            for(int j = 1; j <= Math.sqrt(i); j++) {
                if(i % j == 0) {
                    count++;
                    if(j * j != i) {
                        count++;
                    }
                }
            knights[i - 1] = count;
            }
        }
        
        for(int i = 0; i < knights.length; i++) {
            if(knights[i] > limit) {
                knights[i] = power;
            }
            answer += knights[i];
        }
        
        return answer;
    }
}

}
